  .data( ratData )
  .enter()
  .append( "rect" )
    .attr( "x", function(d,i){
      return i*25;
    })
    .attr( "width", 15 )
    .attr( "fill", "#d1c9b8" )
    .attr( "height", function(d){
      return d/10 * 1.5;
    })
    .attr( "y", function(d){

    });
